## 🐳 DOCKER & KUBERNETES

### Docker Configuration
```dockerfile
# Backend Service
FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production
COPY . .
EXPOSE 3001
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
  CMD node healthcheck.js
CMD ["node", "src/server.js"]

# Frontend Service
FROM node:18-alpine as builder
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
RUN npm run build

FROM nginx:alpine
COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/nginx.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

### Kubernetes Deployment
```yaml
# Backend Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: lernlix-backend
spec:
  replicas: 3
  template:
    spec:
      containers:
      - name: backend
        image: lernlix/backend:1.0.0
        ports:
        - containerPort: 3001
        env:
        - name: DB_HOST
          valueFrom:
            configMapKeyRef:
              name: lernlix-config
              key: db-host
        livenessProbe:
          httpGet:
            path: /api/health
            port: 3001
          initialDelaySeconds: 30
          periodSeconds: 10

# Frontend Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: lernlix-frontend
spec:
  replicas: 2
  template:
    spec:
      containers:
      - name: frontend
        image: lernlix/frontend:1.0.0
        ports:
        - containerPort: 80

# Load Balancer Service
apiVersion: v1
kind: Service
metadata:
  name: lernlix-lb
spec:
  type: LoadBalancer
  selector:
    app: lernlix
  ports:
  - protocol: TCP
    port: 80
    targetPort: 80
```

## 🧪 TESTING FRAMEWORK (ENTERPRISE)

### Unit Testing (Jest)
```javascript
// Example: User Service Tests
describe('UserService', () => {
  describe('registerUser', () => {
    it('should hash password before saving', async () => {
      const user = await UserService.register({
        email: 'test@example.com',
        password: 'plain_password'
      });
      expect(user.password).not.toBe('plain_password');
      expect(await bcrypt.compare('plain_password', user.password)).toBe(true);
    });

    it('should validate email format', async () => {
      expect(() => UserService.register({
        email: 'invalid-email',
        password: 'password123'
      })).toThrow('Invalid email format');
    });

    it('should prevent duplicate emails', async () => {
      await UserService.register({
        email: 'existing@example.com',
        password: 'password123'
      });
      expect(() => UserService.register({
        email: 'existing@example.com',
        password: 'password456'
      })).toThrow('Email already in use');
    });
  });
});
```

### Integration Testing (Supertest)
```javascript
// API Integration Tests
describe('POST /api/auth/login', () => {
  it('should return JWT token for valid credentials', async () => {
    const response = await request(app)
      .post('/api/auth/login')
      .send({
        email: 'user@example.com',
        password: 'password123'
      });

    expect(response.status).toBe(200);
    expect(response.body).toHaveProperty('token');
    expect(response.body.token).toMatch(/^Bearer /);
  });

  it('should return 401 for invalid credentials', async () => {
    const response = await request(app)
      .post('/api/auth/login')
      .send({
        email: 'user@example.com',
        password: 'wrong_password'
      });

    expect(response.status).toBe(401);
    expect(response.body.error).toBe('Invalid credentials');
  });
});
```
